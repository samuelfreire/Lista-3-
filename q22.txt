int imprimv(int el[], int total)
{
int i;
int max = el[0];
for (i = 1; i < total; ++i)
{
if (el[i] > max) max = el[i];
}   
for (int y = max; y >= 1; --y)
{   
for (int x = 0; x < total; ++x)
{
if (el[x] >= y)
printf("*");
else 
printf(" ");
} 
printf("\n"); 
}   
return 0;}
